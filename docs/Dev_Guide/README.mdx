## ğŸ”§ Key Development Patterns

### Svelte 5 Patterns

```typescript
// Reactive state with runes
let count = $state(0);
let doubled = $derived(count * 2);

// Component communication
let { data = $bindable() } = $props();

// Effects and lifecycle
$effect(() => {
	console.log('Count changed:', count);
});
```

### Widget Development Pattern

```typescript
// Widget factory function
export default function MyWidget(config: WidgetConfig) {
	return {
		Name: 'MyWidget',
		component: MyWidgetComponent,
		config,
		Icon: 'mdi:widget',
		Description: 'Custom widget description'
	};
}

// Widget metadata
MyWidget.Name = 'My Widget';
MyWidget.dependencies = ['Input', 'Button'];
```

### API Endpoint Pattern

```typescript
// SvelteKit API route
import type { RequestHandler } from './$types';

export const GET: RequestHandler = async ({ locals, params }) => {
	// Authentication check
	if (!locals.user) {
		throw error(401, 'Unauthorized');
	}

	// Business logic
	const data = await processRequest(params);

	return json({ data });
};
```
